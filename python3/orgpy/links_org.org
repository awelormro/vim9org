Los links son formatos de documento que pueden permitir acceder a una parte
dentro del documento o fuera de él, además de generar ciertos comandos

#+BEGIN_SRC python links_org.py
def go_link(fwd: bool) -> None:
    import vim
    if not fwd:
        verifier = vim.Function('col')(',')
        verifier2 = vim.Function('line')(',')
        vim.Function('search')('\\[\\[', 'b')
        ver_col = vim.Function('col')(',')
        ver_line = vim.Function('line')(',')
        if verifier == ver_col and verifier2 == ver_line:
            vim.Function('cursor')(ver_line, ver_col - 2)
            vim.Function('search')('\\[\\[', 'b')
    else:
        vim.Function('search')('\\[\\[')


#+END_SRC
